분기 = [날짜].[QuarterNo] & '분기'
월No = MONTH([날짜])                                #달에서 날짜를 추출함 
월(영문)= FORMAT([날짜],"mmm")                   #1월 = Jan 세글자로 표현해라   @mmm  
매출금액 = [단가]*[수량]*(1-[할인율])
매출원가 = RELATED('제품'[원가])*[수량]            #related 함수는 다른 테이블의 정보를 가져온다. 
매출이익 = [매출금액] - [매출원가]
총매출금액 = SUM('판매'[매출금액])                  #sum 함수  #측정도구 사용시 따로 테이블에 표시는 안됨
총매출이익 = SUM('판매'[매출이익])
매출이익률 = DIVIDE([총매출이익],[총매출금액], 0) #0은 오류가 나면 0으로 반환하라는 의미 
거래건수 = COUNT('판매'[판매ID])
평균단가 = AVERAGE('판매'[단가])
